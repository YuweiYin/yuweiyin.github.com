def _optimal_bst(self, e_table, w_table, r_table, n):
    assert isinstance(self.p_arr, list) and len(self.p_arr) > 0
    assert isinstance(self.q_arr, list) and len(self.q_arr) == len(self.p_arr) + 1

    # å¯¹äºåä¸ªé¿åº¦(length) ååä¸ªèµ·ç¹(i) çå­å³é®å­åºå(å­é®é¢)æ±æä¼è§£/å¼ï¼èªåºåä¸åå«è®¡ç®
    # - è¥ j == i-1ï¼å e[i, j] = q_{i-1}
    # - è¥ i <= jï¼å e[i, j] = min{ e[i, r-1] + e[r+1, j] + w(i,j) }
    # - w[i, j] = w[i, j-1] + pj + qj
    for length in range(1, n + 1):          # å­å³é®å­åºåçé¿åº¦åå¼èå´ä¸º 1 ~ n
        # ç¬¬ä¸ä¸ªå¾ªç¯
        for i in range(1, n - length + 2):  # å­å³é®å­åºåèµ·ç¹ i åå¼èå´ä¸º 0 ~ n-1
            j = i + length - 1              # å­å³é®å­åºåç»ç¹ j
            e_table[i][j] = self.inf
            w_table[i][j] = round(w_table[i][j - 1] + self.p_arr[j - 1] + self.q_arr[j], self.float_acc)
            for r in range(i, j + 1):
                cost = e_table[i][r - 1] + e_table[r + 1][j] + w_table[i][j]
                if cost < e_table[i][j]:
                    e_table[i][j] = round(cost, self.float_acc)
                    r_table[i][j] = r

    # æ­¤æ¶æä¼å¼ä¸º e[1, n]
    self.optimal_cost = e_table[1][n]
    # è¿åå¤å¿å½ eãw å r
    return e_table, w_table, r_table
